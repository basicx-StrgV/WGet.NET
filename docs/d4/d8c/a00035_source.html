293 public List<WinGetAdminSetting> GetAdminSettings()
294 {
295 List<WinGetAdminSetting> adminSettings = new();
296
297 string settingsJson = ExportSettings();
298
299 SettingsModel settings = JsonHelper.StringToObject<SettingsModel>(settingsJson);
300 if (settings == null)
301 {
302 return adminSettings;
303 }
304
305 WinGetAdminSettingBuilder builder = new();
306 foreach (KeyValuePair<string, bool> entry in settings.AdminSettings)
307 {
308 builder.Clear();
309
310 builder.AddEntryName(entry.Key);
311 builder.AddStatus(entry.Value);
312
313 WinGetAdminSetting? adminSetting = builder.GetInstance();
314 if (adminSetting != null)
315 {
316 adminSettings.Add(adminSetting);
317 }
318 }
319
320 return adminSettings;
321 }
322
336 public async Task<List<WinGetAdminSetting>> GetAdminSettingsAsync(CancellationToken cancellationToken = default)
337 {
338 List<WinGetAdminSetting> adminSettings = new();
339
340 string settingsJson = await ExportSettingsAsync(cancellationToken);
341
342#if NETCOREAPP3_1_OR_GREATER
343 SettingsModel settings = await JsonHelper.StringToObjectAsync<SettingsModel>(settingsJson, cancellationToken);
344#elif NETSTANDARD2_0
345 SettingsModel settings = JsonHelper.StringToObject<SettingsModel>(settingsJson);
346#endif
347 if (settings == null)
348 {
349 return adminSettings;
350 }
351
352 WinGetAdminSettingBuilder builder = new();
353 foreach (KeyValuePair<string, bool> entry in settings.AdminSettings)
354 {
355 if (cancellationToken.IsCancellationRequested)
356 {
357 return adminSettings;
358 }
359
360 builder.Clear();
361
362 builder.AddEntryName(entry.Key);
363 builder.AddStatus(entry.Value);
364
365 WinGetAdminSetting? adminSetting = builder.GetInstance();
366 if (adminSetting != null)
367 {
368 adminSettings.Add(adminSetting);
369 }
370 }
371
372 return adminSettings;

642 public WinGetInfo GetInfo()
643 {
644 ProcessResult result = Execute(_infoCmd);
645
646 InfoActionVersionId actionVersionId = InfoActionVersionId.VersionRange1;
647 if (CheckWinGetVersion(new Version(1, 4, 3531), new Version(1, 5, 101)))
648 {
649 actionVersionId = InfoActionVersionId.VersionRange2;
650 }
651 else if (CheckWinGetVersion(new Version(1, 5, 441), new Version(1, 5, 441)))
652 {
653 actionVersionId = InfoActionVersionId.VersionRange3;
654 }
655 else if (CheckWinGetVersion(new Version(1, 5, 1081)))
656 {
657 actionVersionId = InfoActionVersionId.VersionRange4;
658 }
659
660 return ProcessOutputReader.ToWingetInfo(result.Output, actionVersionId);
661 }

676 public async Task<WinGetInfo> GetInfoAsync(CancellationToken cancellationToken = default)

681 InfoActionVersionId actionVersionId = InfoActionVersionId.VersionRange1;

684 actionVersionId = InfoActionVersionId.VersionRange2;

690 else if (CheckWinGetVersion(new Version(1, 5, 1081)))

694
695 // Return empty data if the task was canceled
696 if (cancellationToken.IsCancellationRequested)
697 {
698 return WinGetInfo.Empty;

701 return ProcessOutputReader.ToWingetInfo(result.Output, actionVersionId);

834 private string CheckWinGetVersion()
835 {
836 if (!IsInstalled)
837 {
838 return string.Empty;
839 }
840
841 ProcessResult result = Execute(_versionCmd);
842
843 for (int i = 0; i < result.Output.Length; i++)
844 {
845#if NETCOREAPP3_1_OR_GREATER
846 if (result.Output[i].StartsWith('v') && result.Output[i].Length >= 2)
847 {
848 // Return output without the 'v' at the start.
849 return result.Output[i].Trim()[1..];
850 }
851#elif NETSTANDARD2_0
852 if (result.Output[i].StartsWith("v") && result.Output[i].Length >= 2)
853 {
854 // Return output without the 'v' at the start.
855 return result.Output[i].Trim().Substring(1);
856 }
857#endif
858 }
859
860 return string.Empty;
861 }

869 private string CheckInstallation()
870 {
871 string? pathEnvVar = Environment.GetEnvironmentVariable("Path", EnvironmentVariableTarget.User);
872 if (string.IsNullOrWhiteSpace(pathEnvVar))
873 {
874 return string.Empty;
875 }
876
877 string[] paths = pathEnvVar.Split(';');
878
879 string exePath;
880 for (int i = 0; i < paths.Length; i++)
881 {
882 exePath = Path.Combine(paths[i], "winget.exe");
883 if (File.Exists(exePath))
884 {
885 return exePath;
886 }
887 }
888
889 return string.Empty;
890 }

914 private bool QueryInstallation()
915 {
916 bool isInstalled;
917
918 _wingetExePath = CheckInstallation();
919
920 if (string.IsNullOrWhiteSpace(_wingetExePath))
921 {
922 isInstalled = false;
923 _processManager = new ProcessManager("winget");
924 }
925 else
926 {
927 isInstalled = true;
928 _processManager = new ProcessManager(_wingetExePath);
929 }
930
931 _wingetExeModificationDate = GetLastModificationData();
932
933 if (isInstalled)
934 {
935 _versionString = CheckWinGetVersion();
936 }
937 else
938 {
939 _versionString = string.Empty;
940 }
941
942 _version = VersionParser.Parse(_versionString);
943
944 _isPreview = VersionParser.CheckPreviewStatus(_versionString);
945
946 return isInstalled;
947 }

956 private bool CheckAdministratorPrivileges()
957 {
958 if (Environment.OSVersion.Platform == PlatformID.Unix)
959 {
960 // Making sure windows related functions dont get called on none windows systems.
961 return true;
962 }
963
964 using WindowsIdentity? identity = WindowsIdentity.GetCurrent(false);
965
966 if (identity != null)
967 {
968 return new WindowsPrincipal(identity)
969 .IsInRole(WindowsBuiltInRole.Administrator);
970 }
971
972 return false;
973 }
