22 public WinGetAdminSettingBuilder()

31 public void AddEntryName(string entryName)
32 {
33 _entryName = entryName;
34 }
35
43 public void AddRawContent(string rawContent)
44 {
45 _hasShortenedContent = CheckShortenedValue(rawContent);
46
47 if (_hasShortenedContent)
48 {
49 // Remove the char at the end of the shortened content.
50 _rawContent = rawContent.Remove(rawContent.Length - 1);
51 }
52 else
53 {
54 _rawContent = rawContent;
55 }
56
57 SetIsEnabled(_rawContent, _hasShortenedContent);
58 }
59
70 public void AddStatus(bool status)
71 {
72 _isEnabled = status;
73
74 // Set the raw content to a value that could be parsed.
75 if (_isEnabled.Value)
76 {
77 _rawContent = "Enabled";
78 }
79 else
80 {
81 _rawContent = "Disabled";
82 }
83 }
84
91 public override WinGetAdminSetting? GetInstance()
92 {
93 if (!_isEnabled.HasValue)
94 {
95 return null;
96 }
97
98 return new WinGetAdminSetting(_entryName, _rawContent, _hasShortenedContent, _isEnabled.Value);
99 }
100
102 public override void Clear()
103 {
104 _entryName = string.Empty;
105 _rawContent = string.Empty;
106 _hasShortenedContent = false;

119 private void SetIsEnabled(string rawContent, bool hasShortenedContent)
120 {
121 _isEnabled = null;
122
123 if (!hasShortenedContent)
124 {
125 switch (rawContent.ToUpper())
126 {
127 case "ENABLED":
128 _isEnabled = true;
129 break;
130 case "DISABLED":
131 _isEnabled = false;
132 break;
133 }
134 }
135 else
136 {
137 // Try to parse data with smalles amount of information possible.
138 // But the value is short enough, so it should not come to this.
139#if NETCOREAPP3_1_OR_GREATER
140 if (rawContent.ToUpper().StartsWith('E'))
141 {
142 _isEnabled = true;
143 }
144 else if (rawContent.ToUpper().StartsWith('D'))
145 {
146 _isEnabled = false;
147 }
148#elif NETSTANDARD2_0
149 if (rawContent.ToUpper().StartsWith("E"))
150 {
151 _isEnabled = true;
152 }
153 else if (rawContent.ToUpper().StartsWith("D"))
154 {
155 _isEnabled = false;
156 }
157#endif
158 }
159 }
160 }
