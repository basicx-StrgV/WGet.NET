92 private void SetDirectoryInfo(string rawContent, bool hasShortenedContent)
93 {
94 try
95 {
96 string path = rawContent;
97
98#if NETCOREAPP3_1_OR_GREATER
99 if (path.StartsWith('%'))
100 {
101 path = Environment.ExpandEnvironmentVariables(path);
102 }
103#elif NETSTANDARD2_0
104 if (path.StartsWith("%"))
105 {
106 path = Environment.ExpandEnvironmentVariables(path);
107 }
108#endif
109
110 if (!hasShortenedContent)
111 {
112 _directoryInfo = new DirectoryInfo(path);
113 }
114
115 // Fallback for an incomplete directory path.
116 _directoryInfo = new DirectoryInfo(TrimLastDirectory(path));
117 }
118 catch

133 private string TrimLastDirectory(string path)

140 lastSeparatorIndex = i;

142 }
143
144 if (lastSeparatorIndex > -1)

146 return path.Remove(lastSeparatorIndex);
