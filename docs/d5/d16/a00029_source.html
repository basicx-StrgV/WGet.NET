108 private ProcessResult RunProcess(ProcessStartInfo processStartInfo)
109 {
110 ProcessResult result = new();
111
112 //Create and run process
113 using (Process proc = new() { StartInfo = processStartInfo })
114 {
115 proc.Start();
116
117 result.Output = ReadSreamOutput(proc.StandardOutput);
118
119 //Wait till end and get exit code
120 proc.WaitForExit();
121
122 // Make sure the process has exited
123 if (!proc.HasExited)
124 {
125 proc.Kill();
126 }
127
128 result.ExitCode = proc.ExitCode;
129 }
130
131 return result;
132 }

147 private async Task<ProcessResult> RunProcessAsync(ProcessStartInfo processStartInfo, CancellationToken cancellationToken = default)
148 {
149 ProcessResult result = new();
150
151 //Create and run process
152 using (Process proc = new() { StartInfo = processStartInfo })
153 {
154 proc.Start();
155
156 result.Output = await ReadSreamOutputAsync(proc.StandardOutput, cancellationToken);
157
158 // Kill the process and return, if the task is canceled
159 if (cancellationToken.IsCancellationRequested && !proc.HasExited)
160 {
161 proc.Kill();
162
163 result.ExitCode = -1;
164
165 return result;
166 }
167
168 //Wait for the processs to exit
169 proc.WaitForExit();
170
171 // Make sure the process has exited
172 if (!proc.HasExited)
173 {
174 proc.Kill();
175 }
176
177 result.ExitCode = proc.ExitCode;
178 }
179
180 return result;
181 }
